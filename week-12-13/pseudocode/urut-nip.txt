procedure urutNIP(in/out tabPeg: ArrPeg)
kamus
  pass, i, temp: integer
  poinIndex, poinPass: integer
algoritma
  while pass <= N do
    pass <- 1
    temp <- tabPeg[pass]

    while i > 0 AND temp.NIP < tabPeg[i-1].NIP do
      tabPeg[i] <- tabPeg[i-1]
      i <- i - 1
    endwhile
    tabPeg[i] <- temp
    pass <- pass + 1
  endwhile
endprocedure
